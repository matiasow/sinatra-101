Get some help

helpers do
  def url_for(name)
    case name
    when :index then '/index'
    when :cart then  '/cart'
    else "/page/#{name}"
  end 
end

url_for :index # => '/index'
url_for :products  # => '/page/products'

Extending Sinatra

module Sinatra 
  module PostGet
    def post_get(route, &block)
      get(route, &block)
      post(route, &block)
    end 
  end
  register PostGet 
end

post_get '/' do
  "Hello #{params[:name]}!"
end

Models: Soul of your swing

Pick your weapon of choice

  • ActiveRecord
  • DataMapper
  • Sequel

Install DataMapper

gem install dm-sqlite-adapter
gem install dm-mysql-adapter
gem install dm-postgres-adapter
gem install dm-oracle-adapter
gem install dm-sqlserver-adapter
gem install data_mapper

Configure

DataMapper.setup :default, 'sqlite:///path/to/database.db'

Define a model

class Event
  include DataMapper::Resource

  property :id, Serial
  property :name, String
  property :event_date, Date
  property :created_at, DateTime
end

Associations

class Event
  has n, :attendants
end

class Attendant
  belongs_to :event
end

Round out

DataMapper.finalize

Migrations

require 'dm-migrations'
DataMapper.auto_migrate!
DataMapper.auto_upgrade!

Use your model

@event = Event.new(:name => 'Ruby Fun Day', :event_date => Date.today)
@event.save

Perform lookups

Event.get(1) # Lookup by id
Event.first(:name => 'Ruby Fun Day') # First element match
Event.last # Last element
Event.all
Event.all(:date => Date.today)

Define scopes elegantly

class Event
  def self.happening_today
    all(:date => Date.today)
  end

  def self.future
    all(:date.gt => Date.today, :order => [ :date.asc ])
  end
end
